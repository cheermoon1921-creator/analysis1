        // 调用 Gemini API (强制输出 JSON 格式)

        async function fetchResearchData(entity, industry, focus, depth) {

            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;

            

            const systemPrompt = `你是一位顶级的商业分析师，曾在麦肯锡、贝恩等顶级咨询公司任职。

你的任务是进行严谨的商业研究。

必须返回严格的 JSON 格式。JSON 的结构必须如下：

{

  "industry_html": "关于行业的宏观分析，包含市场规模、发展趋势、竞争格局等。使用 HTML 标签 (<h3>, <p>, <ul> 等) 排版。",

  "company_html": "关于企业的微观剖析，包含商业模式、核心竞争力、财务状况(预估)或运营现状等。使用 HTML 排版。",

  "brand_html": "关于品牌的资产研究，包含品牌定位、目标受众画像、品牌心智与口碑等。使用 HTML 排版。",

  "insights_html": "战略洞察与未来建议，包含SWOT分析核心点、潜在风险提示及破局建议。使用 HTML 排版。"

}

不要在开头和结尾使用 \`\`\`json 等 markdown 标记，只返回纯净的 JSON 字符串，确保可以被 JSON.parse() 解析。`;


            const userPrompt = `

请分析目标主体：${entity}

所属行业背景：${industry || '请根据主体自行推断'}

分析深度：${depth === 'standard' ? '标准概览' : depth === 'deep' ? '深度战略剖析' : '投资人商业模式与壁垒视角'}

特别关注重点：${focus || '无特殊要求，全面分析即可'}

`;


            let retries = 0;

            const maxRetries = 3;

            

            while (retries < maxRetries) {

                try {

                    const response = await fetch(url, {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify({

                            contents: [{ parts: [{ text: userPrompt }] }],

                            systemInstruction: { parts: [{ text: systemPrompt }] },

                            generationConfig: {

                                responseMimeType: "application/json"

                            }

                        })

                    });

                    

                    if (!response.ok) throw new Error(`HTTP error! status: ${response.status}`);

                    

                    const data = await response.json();

                    const textResult = data.candidates?.[0]?.content?.parts?.[0]?.text;

                    

                    if (!textResult) throw new Error("未获取到分析内容");

                    

                    return JSON.parse(textResult);


                } catch (err) {

                    console.error("API 调用失败:", err);

                    retries++;

                    if (retries >= maxRetries) throw new Error("生成报告失败，请稍后重试。");

                    await new Promise(r => setTimeout(r, Math.pow(2, retries) * 1000));

                }

            }

        }

            // UI 状态更新：加载中

            emptyState.classList.add('hidden');

            tabContents.forEach(c => c.classList.add('hidden')); // 隐藏旧内容

            loadingSkeletons.classList.remove('hidden'); // 显示骨架屏

            document.getElementById('loadingSpinner').classList.remove('hidden');

            document.getElementById('btnText').innerText = "AI 正在深度分析中...";

            generateBtn.disabled = true;

            generateBtn.classList.add('opacity-80', 'cursor-not-allowed');


            try {

                // 发送请求

                document.getElementById('insightsContent').innerHTML = resultJSON.insights_html || "<p>暂无数据</p>";


                showToast("研报生成完毕！");

            } finally {

                // 恢复按钮状态

            const stripHtml = (html) => {

                let tmp = document.createElement("DIV");

                tmp.innerHTML = html;

                // 为了稍微保留一点结构，把 h3 换成 markdown 标题

                const headings = tmp.querySelectorAll('h3');

                const lists = tmp.querySelectorAll('li');

                lists.forEach(l => l.innerText = `- ${l.innerText}`);

                return tmp.innerText || tmp.textContent || "";

            };
